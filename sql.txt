//KORISNICI

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    f_name VARCHAR(50) NOT NULL,
    l_name VARCHAR(50) NOT NULL,
    gender ENUM('male', 'female', 'other') NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    account_id INT NOT NULL UNIQUE,
    birthday DATE NOT NULL,
    country VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    profile_pic VARCHAR(255) DEFAULT NULL
);

CREATE TABLE tickets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ticket_type ENUM('1 day', '3 day', '1 day vip', '3 day vip', '1 day elite', '3 day elite') NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    quantity_available INT NOT NULL
);

DELIMITER //


CREATE TRIGGER set_ticket_price
BEFORE INSERT ON tickets
FOR EACH ROW
BEGIN
    IF NEW.ticket_type = '1 day' THEN
        SET NEW.price = 89.00;
    ELSEIF NEW.ticket_type = '3 day' THEN
        SET NEW.price = 219.00;
    ELSEIF NEW.ticket_type = '1 day vip' THEN
        SET NEW.price = 149.00;
    ELSEIF NEW.ticket_type = '3 day vip' THEN
        SET NEW.price = 369.00;
    ELSEIF NEW.ticket_type = '1 day elite' THEN
        SET NEW.price = 249.00;
    ELSEIF NEW.ticket_type = '3 day elite' THEN
        SET NEW.price = 499.00;
    END IF;
END //

DELIMITER ;

// KOSARICA

CREATE TABLE cart (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    product_id INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    quantity INT NOT NULL,
    size VARCHAR(10) DEFAULT NULL,
    added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
    
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);


ALTER TABLE cart ADD UNIQUE INDEX unique_cart_item (user_id, product_id, size);

 
//NARUDZBE

CREATE TABLE shipped_orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    product_id INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    quantity INT NOT NULL,
    size VARCHAR(10) DEFAULT NULL,
    added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    shipped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
